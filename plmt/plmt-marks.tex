\usepackage{soul}
\setstcolor{del}
\setul{0.2pt}{1pt}
\setulcolor{hlh}

\newcommand\edmark[1]{%
  {\small\color{accent}\bfseries\sffamily #1}%
}

\newcommand\add[2][]{%
\ifstrempty{#1}{%
\textcolor{add}{#2}%
}{%
\textcolor{add}{#2 \edmark{#1}}%
}%
}

\newcommand\remove[2][]{%
\ifstrempty{#1}{%
\textcolor{meta}{\st{#2}}%
}{%
\textcolor{meta}{\st{#2} \edmark{#1}}%
}%
}

\newcommand\note[2][]{%
\ifstrempty{#1}{%
\textcolor{not}{[\small\sffamily #2]}%
}{%
\textcolor{not}{[\small\sffamily #2 -- \edmark{#1}]}%
}%
}

\newcommand\highlight[2][]{%
\ifstrempty{#1}{%
\ul{#2}%
}{%
\ul{#2} (\edmark{#1})%
}%
}
